RETURN
  (
   SELECT Top (@TopN)c.CustomerID,c.CustomerName,sum(GrossAmount) as PurchaseAmount from Equipment_Transaction E
   inner join Customer c on E.CustomerID=c.CustomerID
   where year(TransDate)=@Year
